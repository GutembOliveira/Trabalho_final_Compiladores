// Declaração de constante (não pode ser reatribuída)
const PI = 3.14159;

// Declaração de uma função para calcular a área de um círculo
function calculateArea(radius) {
    // Variável let dentro do escopo da função
    let area = PI * (radius * radius); 
    return area;
}

// Declaração de variável de bloco/escopo (pode ser reatribuída)
let radius = 10;

// Chamada de função e atribuição do resultado
var circleArea = calculateArea(radius);

// Exemplo de uso de if e redeclaração implícita de 'var' (em JS)
if (circleArea > 300) {

    // Atualiza a variável var do escopo global/pai
